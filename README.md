```python
# Your first PyTorch neural network
import torch
import torch.nn as nn

class SimpleNet(nn.Module):
    def __init__(self):
        super().__init__()
        self.layer = nn.Linear(784, 10)
    
    def forward(self, x):
        return self.layer(x)

model = SimpleNet()
print("Neural network created! 🧠")
```
